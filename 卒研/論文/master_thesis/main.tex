   \makeatletter
   \def\mojiparline#1{
        \newcounter{mpl}
        \setcounter{mpl}{#1}
        \@tempdima=\linewidth
        \advance\@tempdima by-\value{mpl}zw
        \addtocounter{mpl}{-1}
        \divide\@tempdima by \value{mpl}
        \advance\kanjiskip by\@tempdima
        \advance\parindent by\@tempdima
   }
   \makeatother
   \def\linesparpage#1{
       \baselineskip=\textheight
       \divide\baselineskip by #1
   }
